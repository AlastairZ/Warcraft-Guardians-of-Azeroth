generic_artifact_rarity_legendary_score = {
	# modifier = { factor = 1 }

	modifier = {
		factor = 0

		has_game_rule = { name = artifact_spawn_rarity value = even_less_random }
		lower_real_tier_than = king
	}

	modifier = {
		factor = 5

		NOT = { has_game_rule = { name = artifact_spawn_rarity value = random } }
		tier = emperor
	}
	modifier = {
		factor = 4

		NOT = { has_game_rule = { name = artifact_spawn_rarity value = random } }
		tier = king
	}
	modifier = {
		factor = 3

		NOT = { has_game_rule = { name = artifact_spawn_rarity value = random } }
		tier = duke
	}
	modifier = {
		factor = 2

		NOT = { has_game_rule = { name = artifact_spawn_rarity value = random } }
		tier = count
	}
}
