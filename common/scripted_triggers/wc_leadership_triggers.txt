allows_archmagus_leader = {
	is_magic_class_trigger = yes
}

allows_offensive_caster_leader = {
	is_magic_class_trigger = yes
}

allows_defensive_caster_leader = {
	is_magic_class_trigger = yes
}

allows_defiler_leader = {
	OR = {
		is_warlock_class_trigger = yes
		is_shadow_priest_class_trigger = yes
		is_necromancer_class_trigger = yes
	}
}

allows_warpriest_leader = {
	OR = {
		is_priest_class_trigger = yes
		is_druid_class_trigger = yes
		is_paladin_class_trigger = yes
		is_loa_priest_class_trigger = yes
	}
}

allows_spell_specialization {
	OR = {
		allows_warpriest_leader = yes
		allows_defiler_leader = yes
		allows_offensive_caster_leader = yes
		allows_defiler_leader = yes
	}
}

has_caster_leader_archetype = {
	OR = {
		trait = offensive_caster_leader
		trait = defensive_caster_leader
		trait = defiler_leader
		trait = warpriest_leader
	}
}

has_unit_leadership_trait_trigger = {
	OR = {
		# Warcraft
		trait = aerial_raider_leader
		trait = marksman_captain_leader
		trait = artillery_engineer_leader

		trait = cavalry_leader
		trait = heavy_infantry_leader
		trait = light_foot_leader
	}
}
