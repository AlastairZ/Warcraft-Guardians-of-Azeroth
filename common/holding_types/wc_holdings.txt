trade_post = {
	trigger = {

		### Land

		trigger_if = {
			limit = { trade_route = human_trade }
			can_build_trade_post_trigger = yes
			patrician_condition_trigger = yes
			custom_tooltip = {
				text = inHumantradepost
				northern_eastern_kingdoms_trade_post_trigger = yes
			}
		}
		trigger_else_if = {
			limit = { trade_route = azeroth_trade }
			can_build_trade_post_trigger = yes
			patrician_condition_trigger = yes
			custom_tooltip = {
				text = inAzerothiantradepost
				south_eastern_kingdoms_trade_post_trigger = yes
			}
		}
		trigger_else_if = {
			limit = { trade_route = goldenroad_trade }
			can_build_trade_post_trigger = yes
			patrician_condition_trigger = yes
			custom_tooltip = {
				text = inKalimdortradepost
				kalimdor_trade_post_trigger = yes
			}
		}

		### Sea

		trigger_else_if = {
			limit = { trade_route = eastern_sea_kalimdor_route }
			can_build_trade_post_trigger = yes
			patrician_condition_trigger = yes
			custom_tooltip = {
				text = eastern_sea_kalimdor_route_tooltip
				eastern_sea_kalimdor_route_trigger = yes
			}
		}

		trigger_else_if = {
			limit = { trade_route = great_sea_trade_route }
			can_build_trade_post_trigger = yes
			patrician_condition_trigger = yes
			custom_tooltip = {
				text = great_sea_trade_route_tooltip
				great_sea_trade_route_trigger = yes
			}
		}

		trigger_else_if = {
			limit = { trade_route = eastern_kingdoms_sea_route }
			can_build_trade_post_trigger = yes
			patrician_condition_trigger = yes
			custom_tooltip = {
				text = eastern_kingdoms_sea_route_tooltip
				eastern_kingdoms_sea_route_trigger = yes
			}
		}
	}
}
