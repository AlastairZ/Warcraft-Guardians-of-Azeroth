	random_list = { # Prefix
		1 = { set_artifact_flag = temp_artifact_prefix_flag_1 }
		1 = { set_artifact_flag = temp_artifact_prefix_flag_2 }
		1 = { set_artifact_flag = temp_artifact_prefix_flag_3 }
		1 = { set_artifact_flag = temp_artifact_prefix_flag_4 }
		1 = { set_artifact_flag = temp_artifact_prefix_flag_5 }
		1 = { set_artifact_flag = temp_artifact_prefix_flag_6 }
		1 = { set_artifact_flag = temp_artifact_prefix_flag_7 }
		1 = { set_artifact_flag = temp_artifact_prefix_flag_8 }
		1 = { set_artifact_flag = temp_artifact_prefix_flag_9 }
		1 = { set_artifact_flag = temp_artifact_prefix_flag_10 }
	}
	random_list = { # Suffix
		1 = { set_artifact_flag = temp_artifact_suffix_flag_1 }
		1 = { set_artifact_flag = temp_artifact_suffix_flag_2 }
		1 = { set_artifact_flag = temp_artifact_suffix_flag_3 }
		1 = { set_artifact_flag = temp_artifact_suffix_flag_4 }
		1 = { set_artifact_flag = temp_artifact_suffix_flag_5 }
		1 = { set_artifact_flag = temp_artifact_suffix_flag_6 }
		1 = { set_artifact_flag = temp_artifact_suffix_flag_7 }
		1 = { set_artifact_flag = temp_artifact_suffix_flag_8 }
		1 = { set_artifact_flag = temp_artifact_suffix_flag_9 }
		1 = { set_artifact_flag = temp_artifact_suffix_flag_10}
	}
}

finish_artifact_creation_effect = {
	set_creation_date = 1.1.1
	clear_flags_with_prefix = temp_artifact_
}
