set_prefix_and_suffix_flags_to_weapon_artifact_effect = {
	set_prefix_10_temp_flags_artifact_effect = yes
	if = {
		limit = {
			NOR = {
				has_artifact_flag = artifact_weapon_element_fire_flag
				has_artifact_flag = artifact_weapon_element_frost_flag
				has_artifact_flag = artifact_weapon_element_lightning_flag
				has_artifact_flag = artifact_weapon_element_poison_flag
				has_artifact_flag = artifact_weapon_element_magic_flag
				has_artifact_flag = artifact_weapon_element_physical_flag
			}
		}
		random_list = {
			3 = { set_artifact_flag = artifact_weapon_element_fire_flag			}
			3 = { set_artifact_flag = artifact_weapon_element_frost_flag		}
			3 = { set_artifact_flag = artifact_weapon_element_lightning_flag	}
			3 = { set_artifact_flag = artifact_weapon_element_poison_flag		}
			3 = { set_artifact_flag = artifact_weapon_element_magic_flag		}
			8 = { set_artifact_flag = artifact_weapon_element_physical_flag		}
		}
	}
	random_list = { # Suffix
		1 = { set_artifact_flag = temp_artifact_suffix_flag_1 }
		1 = { set_artifact_flag = temp_artifact_suffix_flag_2 }
		1 = { set_artifact_flag = temp_artifact_suffix_flag_3 }
		1 = { trigger = { has_artifact_flag = artifact_weapon_element_physical_flag } set_artifact_flag = temp_artifact_suffix_flag_4 }
		1 = { trigger = { has_artifact_flag = artifact_weapon_element_physical_flag } set_artifact_flag = temp_artifact_suffix_flag_5 }
		1 = { trigger = { has_artifact_flag = artifact_weapon_element_physical_flag } set_artifact_flag = temp_artifact_suffix_flag_6 }

	}

}

give_new_artifact_random_name_and_origin_effect = {
	save_new_artifact_as_event_target_effect = yes
	event_target:target_new_artifact_generic = {
		if = {
			limit = { quality == 1 }
			set_description = "artifact_common_origin"
		}
		else_if = {
			limit = { quality == 2 }
			set_description = "artifact_uncommon_origin"
		}
		else_if = {
			limit = { quality == 3 }
			set_description = "artifact_rare_origin"
		}
		else_if = {
			limit = { quality == 4 }
			set_description = "artifact_epic_origin"
		}
		else = { set_description = "artifact_legendary_origin" }
		give_artifact_random_name_effect = yes
		set_picture_to_weapon_artifact_effect = yes
		set_creation_date = 1.1.1
	}
	finish_artifact_creation_effect = yes
}
