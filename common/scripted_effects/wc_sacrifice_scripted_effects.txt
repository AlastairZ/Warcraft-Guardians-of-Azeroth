caltulate_PREV_offmap_old_gods_currency_insanity_effect = {
	PREV = {
		show_scope_change = no

		if = {
			limit = {
				OR = {
					cthun_offmap_decisions_enabled_trigger = yes
					nzoth_offmap_decisions_enabled_trigger = yes
					yogg_saron_offmap_decisions_enabled_trigger = yes
					yshaarj_offmap_decisions_enabled_trigger = yes
				}
			}
			custom_tooltip = {
				text = gain_old_god_offmap_currency_tooltip

				set_variable = { which = received_offmap_old_gods_currency_insanity_variable value = 100 }
			}
		}
	}
	else = { break = yes }
	hidden_tooltip = {
		# Priests
		if = {
			limit = { controls_religion = yes }
			PREV = { change_variable = { which = received_offmap_old_gods_currency_insanity_variable value = 500 } }
		}
		if = {
			limit = {
				OR = {
					is_priest = yes
					is_ascetic_trigger = yes
				}
			}
			PREV = { change_variable = { which = received_offmap_old_gods_currency_insanity_variable value = 100 } }
		}
		# Rulers and tiers
		if = {
			limit = { is_ruler = yes }
			PREV = { change_variable = { which = received_offmap_old_gods_currency_insanity_variable value = 50 } }

			if = {
				limit = { tier = COUNT }
				PREV = { change_variable = { which = received_offmap_old_gods_currency_insanity_variable value = 50 } }
			}
			else_if = {
				limit = { tier = DUKE }
				PREV = { change_variable = { which = received_offmap_old_gods_currency_insanity_variable value = 150 } }
			}
			else_if = {
				limit = { tier = KING }
				PREV = { change_variable = { which = received_offmap_old_gods_currency_insanity_variable value = 300 } }
			}
			else = {
				limit = { tier = EMPEROR }
				PREV = { change_variable = { which = received_offmap_old_gods_currency_insanity_variable value = 500 } }
			}
		}
		# Child
		if = {
			limit = { is_adult = no }
			PREV = { change_variable = { which = received_offmap_old_gods_currency_insanity_variable value = 100 } }
		}
		# Has the blood of Twilight Viceroy
		if = {
			limit = { PREV = { any_owned_bloodline = { has_bloodline_flag = bloodline_insanity_gain } } }
			PREV = { multiply_variable = { which = received_offmap_old_gods_currency_insanity_variable value = 1.5 } }
		}
		# C'thun likes/dislikes
		if = {
			limit = { PREV = { cthun_offmap_decisions_enabled_trigger = yes } }

			if = {
				limit = { disliked_by_offmap = { type = offmap_cthun } }
				PREV = { multiply_variable = { which = received_offmap_old_gods_currency_insanity_variable value = 1.25 } }
			}
			if = {
				limit = { PREV = { liked_by_offmap = { type = offmap_cthun } } }
				PREV = { multiply_variable = { which = received_offmap_old_gods_currency_insanity_variable value = 1.25 } }
			}
			else_if = {
				limit = { PREV = { disliked_by_offmap = { type = offmap_cthun } } }
				PREV = { multiply_variable = { which = received_offmap_old_gods_currency_insanity_variable value = 0.8 } }
			}
		}
		# N'zoth likes/dislikes
		if = {
			limit = { PREV = { nzoth_offmap_decisions_enabled_trigger = yes } }

			if = {
				limit = { disliked_by_offmap = { type = offmap_nzoth } }
				PREV = { multiply_variable = { which = received_offmap_old_gods_currency_insanity_variable value = 1.25 } }
			}
			if = {
				limit = { PREV = { liked_by_offmap = { type = offmap_nzoth } } }
				PREV = { multiply_variable = { which = received_offmap_old_gods_currency_insanity_variable value = 1.25 } }
			}
			else_if = {
				limit = { PREV = { disliked_by_offmap = { type = offmap_nzoth } } }
				PREV = { multiply_variable = { which = received_offmap_old_gods_currency_insanity_variable value = 0.8 } }
			}
		}
		# Yogg-Saron likes/dislikes
		if = {
			limit = { PREV = { yogg_saron_offmap_decisions_enabled_trigger = yes } }

			if = {
				limit = { disliked_by_offmap = { type = offmap_yoggsaron } }
				PREV = { multiply_variable = { which = received_offmap_old_gods_currency_insanity_variable value = 1.25 } }
			}
			if = {
				limit = { PREV = { liked_by_offmap = { type = offmap_yoggsaron } } }
				PREV = { multiply_variable = { which = received_offmap_old_gods_currency_insanity_variable value = 1.25 } }
			}
			else_if = {
				limit = { PREV = { disliked_by_offmap = { type = offmap_yoggsaron } } }
				PREV = { multiply_variable = { which = received_offmap_old_gods_currency_insanity_variable value = 0.8 } }
			}
		}

		# Actual Effect
		if = {
			limit = { PREV = { cthun_offmap_decisions_enabled_trigger = yes } }
			PREV = {
				add_offmap_currency = {
					offmap = offmap_cthun
					value = received_offmap_old_gods_currency_insanity_variable
				}
			}
		}
		else_if = {
			limit = { PREV = { nzoth_offmap_decisions_enabled_trigger = yes } }
			PREV = {
				add_offmap_currency = {
					offmap = offmap_nzoth
					value = received_offmap_old_gods_currency_insanity_variable
				}
			}
		}
		else_if = {
			limit = { PREV = { yogg_saron_offmap_decisions_enabled_trigger = yes } }
			PREV = {
				add_offmap_currency = {
					offmap = offmap_yoggsaron
					value = received_offmap_old_gods_currency_insanity_variable
				}
			}
		}

		PREV = { set_variable = { which = received_offmap_old_gods_currency_insanity_variable value = 0 } }

	}
}
