purge_patron_character_effect = {
	if = {
		limit = {
			NOT = {
				has_game_rule = {
					name = patrons_game_rule
					value = patron_trait
				}
			}
		}

		if = { limit = { trait = patron		} remove_trait = patron		}
		if = { limit = { trait = fan_trait	} remove_trait = fan_trait	}

		if = {
			limit = {
				has_game_rule = {
					name = patrons_game_rule
					value = purged
				}
				ai = yes
				NOT = { any_close_relative = { ai = no } }	# Cait-eriona fix
				NOT = { any_spouse = { ai = no } }			# Rei fix
			}
			death={death_reason=death_purged}
		}
	}
}
