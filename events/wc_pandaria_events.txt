### The Mists of Pandaria parted
# on_yearly_pulse
#Pinger event
character_event = {
	id = WCPAN.99
	
	is_triggered_only = yes
	hide_window = yes
	
	ai = no
	
	trigger = {
		OR = {
			multiplayer = no
			is_multiplayer_host_character = yes
		}
		
		NOT = { has_game_rule = { name = mists_of_pandaria value = off } }
		
		year = 613
	}
	
	immediate = {
		set_global_flag = mists_of_pandaria_parted_flag
		
		any_player = {
			narrative_event = { id = WCPAN.100 }
		}
	}
}
#Ping event

# Pandaria Empire is formed
narrative_event = {
	id = WCPAN.200
	title = EVTTITLE_WCPAN_200
	desc = EVTDESC_WCPAN_200
	picture = GFX_evt_pandaria_twin_monoliths
	border = GFX_event_narrative_frame_diplomacy

	is_triggered_only = yes

	# I am the heart of Pandaria!
	option = {
		trigger = {
			character = event_target:target_emperor
		}
		name = EVTOPTA_WCPAN_200
	}
	option = {
		trigger = {
			NOT = {
				character = event_target:target_emperor
			}
			religion_group = event_target:target_emperor
			culture_group = event_target:target_emperor
		}
		name = EVTOPTB_WCPAN_200
	}
	option = {
		trigger = {
			NOT = {
				character = event_target:target_emperor
				AND = {
					religion_group = event_target:target_emperor
					culture_group = event_target:target_emperor
				}
			}
		}
		name = INTERESTING
	}
	option = {
		trigger = {
			NOT = {
				character = event_target:target_emperor
				AND = {
					religion_group = event_target:target_emperor
					culture_group = event_target:target_emperor
				}
			}
		}
		name = ALAS
	}
}
