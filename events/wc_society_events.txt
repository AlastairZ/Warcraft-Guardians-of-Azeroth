#Finish whisphers martial
character_event = {
	id = WCCSO.10206
	desc = EVTDESC_WCCSO.10206
	picture = GFX_evt_secluded_in_prayer
	border = GFX_event_normal_frame_religion

	is_triggered_only = yes

	option = { #Awesome!
		name = EVTOPTA_WCCSO.10206
		if = {
			limit = {
				NOT = { learning = 12 }
			}
			piety = 50
    		add_artifact = whisphers_martial_1
		}
		if = {
			limit = {
				AND = {
					learning = 12
					NOT = { learning = 18 }
				}
			}
			piety = 100
    		add_artifact = whisphers_martial_2
		}
		if = {
			limit = {
				AND = {
					learning = 18
					NOT = { learning = 25 }
				}
			}
			piety = 150
    		add_artifact = whisphers_martial_3
		}
		if = {
			limit = {
				learning = 25
			}
			piety = 200
    		add_artifact = whisphers_martial_4
		}
		remove_character_modifier = writing_magnum_opus
	}
}
