	id = WCGW.1
	desc = GWCBDESC_1
	picture = GFX_evt_gnoll_packlord
	border = GFX_event_normal_frame_war

	is_triggered_only = yes

  option = { #Gain Garfang bloodline.
    name = "EVTGW1"
    create_bloodline = { type = garfang }
    if = {
      limit = { has_nickname = no }
      give_nickname = nick_the_great_beast
    }
  }
}
