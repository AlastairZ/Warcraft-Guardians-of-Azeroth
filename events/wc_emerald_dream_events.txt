namespace=WCED

character_event = {
	id = WCED.1
	desc = EVTDESC_WCED.1
	picture = GFX_evt_emerald_dream
	border = GFX_event_normal_frame_religion

	is_triggered_only = yes

	immediate = {
	}

	option = {
		name = EVTOPTA_WCED.1
		add_trait = in_emerald_dream
		if = {
			limit = {
				has_quest = enter_emerald_dream
			}
			clr_quest = enter_emerald_dream
			add_society_currency_tiny_effect = yes
		}
		if = {
			limit = {
				has_character_modifier = barrow_den_small
			}
			add_character_modifier = barrow_den_weak_protection
		}
		if = {
			limit = {
				has_character_modifier = barrow_den_medium
			}
			add_character_modifier = barrow_den_average_protection
		}
		if = {
			limit = {
				has_character_modifier = barrow_den_large
			}
			add_character_modifier = barrow_den_good_protection
		}
		set_character_flag = in_the_dream
		clr_character_flag = recently_visited_dream
	}
	
	option = {
		name = EVTOPTB_WCED.1
	}
}
