# The Order asks permission to build a castle on your land
character_event = {
	id = WCORD.9000

	is_triggered_only = yes
	hide_window = yes
	
	only_rulers = yes

	trigger = {
		is_alive = yes
		is_dying = no
		
		holy_order = yes
		wealth = 300
		NOT = { realm_size = 10 }
		NOT = { num_of_count_titles = 1 }

		NOT = { has_landed_title = d_nethergarde }

		any_playable_ruler = {
			is_untouchable_trigger = no
			can_rule_peacefully_trigger = yes
			
			is_tribal = no
			OR = {
				opinion = { who = ROOT value = 5 }
				religion = ROOT
			}
			any_demesne_province = {
				has_empty_holding = yes
			}
		}
	}

	weight_multiplier = {
		days = 1

		modifier = {
			factor = 2
			NOT = { demesne_size = 1 }
		}
		modifier = {
			factor = 2
			NOT = { demesne_size = 2 }
		}
		modifier = {
			factor = 0.5
			realm_size = 5
		}
	}

	immediate = {
		random_playable_ruler = {
			limit = {
				is_untouchable_trigger = no
				can_rule_peacefully_trigger = yes
				
				is_tribal = no
				OR = {
					opinion = { who = ROOT value = 5 }
					religion = ROOT
				}
			}
			random_demesne_province = {
				limit = { has_empty_holding = yes }
				save_event_target_as = target_castle_location
			}
			letter_event = { id = WCORD.9015 }
		}
	}
}
