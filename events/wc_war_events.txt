# Spawns coalition leader
character_event = {
	id = WCWAR.600
	
	is_triggered_only = yes
	hide_window = yes
	
	ai = no
	
	trigger = {
		OR = {
			multiplayer = no
			is_multiplayer_host_character = yes
		}
		
		NOT = {
			has_game_rule = {
				name = stop_invader_coalition
				value = off
			}
		}
	}
	
	immediate = {
		log = "[Root.GetTitledFirstName] starts coalition"
		
		capital_scope = { save_event_target_as = target_province }
		
		spawn_random_blank_character_effect = yes
		new_character = {
			save_event_target_as = target_coalition_leader
			
			remove_education_effect = yes
			add_trait = brilliant_strategist
			
			diplomatic_immunity = yes
			set_character_flag = special_marshal 				# Always allow to be in command
			set_character_flag = ai_flag_refuse_conversion
			
			create_title = {
				tier = DUKE
				
				name = "AZEROTH_COALITION"
				ruler = "HIGHLORD_TITLE"
				ruler_female = "HIGHLORD_FEM_TITLE"
				
				holder = THIS
				
				culture = THIS
				
				custom_created = yes
				landless = yes
				short_name = yes
				# temporary = yes
			}
			new_title = {
				set_title_flag = coalition_flag					# Used in CBs
			}
			
			# Adds race trait
			character_event = { id = WCRAC.10 }
			
			# Picks war target
			character_event = { id = WCWAR.610 }
		}
	}
}
# Picks war target
character_event = {
	id = WCWAR.610
	
	is_triggered_only = yes
	hide_window = yes
	
	immediate = {
		# Tries to declare a war...
		if = {
			limit = {
				any_independent_ruler = {
					OR = {
						ai = yes
						has_game_rule = {
							name = stop_invader_coalition
							value = unrestricted
						}
					}
					
					# realm_size = 150
					NAND = {				# Since this invasioin isn't long lasting
						has_landed_title = e_horde
						has_global_flag = orcish_invasion_flag
					}
					event_target:target_coalition_leader = {
						can_use_cb = {
							target = PREV
							casus_belli = stop_invader
							only_check_triggers = yes
						}
					}
				}
			}
			
			# Let's give him gold, prestige and piety
			wealth = 1000
			prestige = 1000
			piety = 500
			
			# Spawns a few commanders
			spawn_fantastic_commander_effect = yes
			spawn_great_commander_effect = yes
			spawn_great_commander_effect = yes
			spawn_good_commander_effect = yes
			spawn_good_commander_effect = yes
			
			random_independent_ruler = {
				limit = {
					OR = {
						ai = yes
						has_game_rule = {
							name = stop_invader_coalition
							value = unrestricted
						}
					}
					
					# realm_size = 150
					NAND = {				# Since this invasioin isn't long lasting
						has_landed_title = e_horde
						has_global_flag = orcish_invasion_flag
					}
					event_target:target_coalition_leader = {
						can_use_cb = {
							target = PREV
							casus_belli = stop_invader
							only_check_triggers = yes
						}
					}
				}
				save_event_target_as = target_invader
				
				# Picks province to spawn army
				if = {			# Capital of invader's neighbor
					limit = {
						any_neighbor_independent_ruler = { is_landed = yes }
					}
					random_neighbor_independent_ruler = {
						limit = {
							is_landed = yes
						}
						capital_scope = {
							save_event_target_as = target_province
						}
					}
				}
				else_if = {		# Coast province of invader
					limit = {
						any_realm_province = {
							port = yes
						}
					}
					random_realm_province = {
						limit = {
							port = yes
						}
						save_event_target_as = target_province
					}
				}
				
				reverse_unsafe_war = {
					casus_belli = stop_invader
					target = event_target:target_coalition_leader
				}
			}
			
			# Spawns army
			spawn_coalition_troops_effect = yes
			
			# Invites other rulers...
			event_target:target_invader = {
				any_neighbor_independent_ruler = {						# Invites independent neighbors...
					invite_to_coalition_war_effect = yes
					any_neighbor_independent_ruler = {						# ...and neighbors of independent neighbors...
						invite_to_coalition_war_effect = yes
					}
				}
			}
		}
		# There's no war target, so let him peacefully die
		else = {
			character_event = { id = WCWAR.700 } # Cleanup
			death = { death_reason = death_missing }
		}
	}
}
