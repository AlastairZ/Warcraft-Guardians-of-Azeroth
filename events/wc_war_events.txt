# Create new troops
character_event = {
	id = WCWAR.471

	hide_window = yes
	is_triggered_only = yes

	immediate = {
		log = "WCWAR.471 fired for [Root.GetBestName]"
		if = {
			limit = {
				NOT = { prestige = 375 }
				NOT = { wealth = 150 }
			}
			log = "WCWAR.471 failed initial trigger for [Root.GetBestName]"
			break = yes
		}
		location = { log = "WCWAR.471 ROOT is [Root.GetBestName], ROOT's location is [This.GetName] last_province is [last_province.GetName]" }
		if = {
			limit = {
				prestige = 375
			}
			spawn_unit = {
				owner = ROOT
				province = event_target:last_province
				home = event_target:last_province
				troops = {
					light_infantry = { 50 50 }
					heavy_infantry = { 125 125 }
					light_cavalry = { 20 20 }
					knights = { 5 5 }
					archers = { 50 50 }
				}
				attrition = 1.0
				is_looter = yes
				can_toggle_looting = no
				reinforces = yes
				merge = yes
				maintenance = no
			}
			prestige = -125
		}
		if = {
			limit = {
				wealth = 150
			}
			spawn_unit = {
				owner = ROOT
				province = event_target:last_province
				home = event_target:last_province
				troops = {
					light_infantry = { 50 50 }
					heavy_infantry = { 125 125 }
					light_cavalry = { 20 20 }
					knights = { 5 5 }
					archers = { 50 50 }
				}
				attrition = 1.0
				is_looter = yes
				can_toggle_looting = no
				reinforces = yes
				merge = yes
				maintenance = no
			}
			wealth = -50
		}
		repeat_event = { id = WCWAR.471 }
	}
}

# Returns the Blackrock tribe back to the Searing Gorge after a defeat
character_event = {
	id = WCWAR.500
	desc = "EVTDESC_WCWAR_500"
	picture = "GFX_evt_grom_01"
	border = "GFX_event_normal_frame_war"

	is_triggered_only = yes

	trigger = {
		is_landed = no
		independent = yes
		war = no

		NOT = { has_global_flag = WCWAR_500_flag }

	}

		set_global_flag = WCWAR_500_flag
		c_blackrock_spire = { add_claim = ROOT }
		d_burning_steppes = { any_direct_de_jure_vassal_title = { add_claim = ROOT } }

		name = LOKTAR_OGAR
		c_blackrock_spire = {
			holder_scope = {
				top_liege = {
					ROOT = {
						if = {
							limit = {
								total_claims = {
									who = PREV

									value = 2
								}
							}
							unsafe_war = {
								target = PREV
								casus_belli = claim_all

							}
						}

							unsafe_war = {
								target = PREV
								casus_belli = claim

							}
						}
					}
				}
			}
		}
		spawn_unit = {
			owner = ROOT
			province = 128	# Dustwind
			home = 128		# Dustwind
			troops = {
				archers = { 200 200 }

			}
			attrition = 0
			maintenance = no
			earmark = horde_forces
		}
	}
	option = {
		name = EVTOPTB_WCWAR_500
		ai_chance = {
			factor = 0
		}
	}
}

### Stop Invader Coalition
# Spawns coalition leader
character_event = {
	id = WCWAR.600
	
	is_triggered_only = yes
	hide_window = yes
	
	ai = no
	
	trigger = {
		OR = {
			multiplayer = no
			is_multiplayer_host_character = yes
		}
		
		NOT = {
			has_game_rule = {
				name = stop_invader_coalition
				value = off
			}
		}
	}
	
	immediate = {
		capital_scope = { save_event_target_as = target_province }
		
		create_character = {
			random_traits = yes
			dynasty = none
			religion = holy_light
			culture = lordaeronian
			female = no
			age = 25
			attributes = {
				martial = 5
			}
			trait = brilliant_strategist
		}
		new_character = {
			save_event_target_as = target_coalition_leader
			
			diplomatic_immunity = yes
			set_character_flag = special_marshal 				# Always allow to be in command
			set_character_flag = coalition_leader_flag		# Used in CBs
			set_character_flag = ai_flag_refuse_conversion
			
			create_title = {
				tier = DUKE
				landless = yes
				temporary = yes
				short_name = yes
				culture = lordaeronian
				name = "AZEROTH_COALITION"
				ruler = "HIGHLORD_TITLE"
				ruler_female = "HIGHLORD_FEM_TITLE"
				holder = THIS
			}
			
			# Picks war target
			character_event = { id = WCWAR.610 }
		}
	}
}
# Picks war target
character_event = {
	id = WCWAR.610
	
	is_triggered_only = yes
	hide_window = yes
	
	immediate = {
		# Tries to declare a war...
		if = {
			limit = {
				any_independent_ruler = {
					OR = {
						ai = yes
						has_game_rule = {
							name = stop_invader_coalition
							value = unrestricted
						}
					}
					
					event_target:target_coalition_leader = {
						can_use_cb = {
							target = PREV
							casus_belli = stop_invader
							only_check_triggers = yes
						}
					}
				}
			}
			
			# Let's give him gold, prestige and piety
			wealth = 1000
			prestige = 1000
			piety = 500
			
			# Spawns a few commanders
			spawn_fantastic_commander_effect = yes
			spawn_great_commander_effect = yes
			spawn_great_commander_effect = yes
			spawn_good_commander_effect = yes
			spawn_good_commander_effect = yes
			
			# Spawns army
			spawn_coalition_troops_effect = yes
			
			random_independent_ruler = {
				limit = {
					OR = {
						ai = yes
						has_game_rule = {
							name = stop_invader_coalition
							value = unrestricted
						}
					}
					
					event_target:target_coalition_leader = {
						can_use_cb = {
							target = PREV
							casus_belli = stop_invader
							only_check_triggers = yes
						}
					}
				}
				save_event_target_as = target_invader
				
				reverse_unsafe_war = {
					casus_belli = stop_invader
					target = event_target:target_coalition_leader
				}
			}
			
			# Invites other rulers...
			event_target:target_invader = {
				any_neighbor_independent_ruler = {						# Invites independent neighbors...
					invite_to_coalition_war_effect = yes
					any_neighbor_independent_ruler = {						# ...and neighbors of independent neighbors...
						invite_to_coalition_war_effect = yes
					}
				}
			}
		}
		# There's no war target, so let him peacefully die
		else = {
			character_event = { id = HL.199 } # Cleanup
			death = { death_reason = death_missing }
		}
	}
}
