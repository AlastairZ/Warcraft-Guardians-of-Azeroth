decisions = {
		repay_loan = {
		  is_high_prio = yes # Will show an alert once decision is enabled
		  
		  ai_check_interval = 12 # Determines how seldom the AI checks whether they can and should do the decision, in months. By default 1 (every month).
		  
		  potential = {
			OR = {
				has_character_modifier = small_debt
				has_character_modifier = average_debt
				has_character_modifier = massive_debt
			}
		  }
		 
		 allow = {
			OR = {
				AND = {
					has_character_modifier = small_debt
					wealth = 150
				}
				AND = {
					has_character_modifier = average_debt
					wealth = 350
				}
				AND = {
					has_character_modifier = massive_debt
					wealth = 800
				}
			}
		  }
		 
		 effect = {
			if = {
			  limit = {
					has_character_modifier = small_debt
			  }
				wealth = -150
				remove_character_modifier = small_debt
			}
			if = {
			  limit = {
					has_character_modifier = average_debt
			  }
				wealth = -350
				remove_character_modifier = average_debt
			}
			if = {
			  limit = {
					has_character_modifier = massive_debt
			  }
				wealth = -800
				remove_character_modifier = massive_debt
			}			
		  }
		 
		 ai_will_do = {
			factor = 1
			modifier = {
				factor = 0
				war = yes
			}
		  }
	  }
}
